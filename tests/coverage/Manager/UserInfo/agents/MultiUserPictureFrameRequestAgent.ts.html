import { BaseRequestAgent, LRUCache } from "./BaseRequestAgent";
import { RetryStrategy } from "../utils/RetryStrategy";
import { PendingRequestTracker } from "../utils/PendingRequestTracker";
import { api } from "../../../Server/Http/Api";
import { UserPictureFrameRequest } from "../types";
import { userInfoMgr } from "../UserInfoManager";
 
/**
 * 批量用户头像框请求代理
 */
export class MultiUserPictureFrameRequestAgent extends BaseRequestAgent<UserPictureFrameRequest, number> {
    protected getAgentName(): string {
        return "MultiUserPictureFrameRequestAgent";
    }
 
    protected getUids(): number[] {
        return this.requests.map(request => request.uid);
    }
 
    protected isUseCacheForRequest(index: number): boolean {
        return this.requests[index].isUseCache;
    }
 
    protected getCache(): LRUCache<number> {
        return userInfoMgr.getPictureFrameLRUCache();
    }
 
    protected getTracker(): PendingRequestTracker {
        return userInfoMgr['_pendingFrameTracker'];
    }
 
    protected getTrackerKey(uid: number): string {
        return `frame-${uid}`;
    }
 
    protected fetchBatch(uids: number[]): Promise<any> {
        return api.getMultiPictureFrames(uids);
    }
 
    protected extractItems(data: any): number[] {
        return data.picture_frames || [];
    }
 
    protected getItemUid(item: number): number {
        // 头像框数据没有uid字段，需要通过索引匹配
        // 这里返回-1表示需要特殊处理
        return -1;
    }
 
    /**
     * 重写批次响应处理逻辑
     * 因为头像框API返回格式特殊：{uids: number[], picture_frames: number[]}
     */
    protected processBatchResponse(
        data: any,
        currentBatch: number[],
        result: Map<number, number>,
        cacheLRU: LRUCache<number>,
        missingIndexMap: Map<number, number>,
    ): void {
        const frames = data.picture_frames || [];
        const requestUids = data.uids || [];
 
        // 按索引匹配uid和frame
        for (let i = 0; i < requestUids.length && i < frames.length; i++) {
            const uid = requestUids[i];
            const pictureFrame = frames[i];
            const originalIndex = missingIndexMap.get(uid);
 
            if (uid !== null && uid !== undefined && originalIndex !== undefined && pictureFrame !== undefined) {
                cacheLRU.set(uid, pictureFrame);
                result.set(uid, pictureFrame);
            }
        }
    }
 
    protected executeCallbacks(result: Map<number, number>): void {
        this.requests.forEach((request) => {
            const pictureFrame = result.get(request.uid);
            const callback = request.callback;
 
            if (!callback) return;
 
            // 检测回调类型：如果回调函数参数长度为1，则视为新的统一回调格式
            // 否则使用旧的回调格式（向后兼容）
            if (callback.length === 1 && callback.toString().includes('result')) {
                // 新格式：UnifiedCallback<number>
                const requestResult = this.buildRequestResult(request.uid, pictureFrame);
                (callback as any)(requestResult);
            } else {
                // 旧格式：UserPictureFrameCallback
                callback(pictureFrame);
            }
        });
    }
}
